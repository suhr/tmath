\DeclareRobustCommand{\rvdots}{%
  \vbox{
    \baselineskip4\p@\lineskiplimit\z@
    \kern-\p@
    \addvspace{2pt}
    \hbox{.}\hbox{.}\hbox{.}
    \addvspace{0.5pt}
  }
}
\makeatother

    \def\fCenter{\mbox{$\;\vdash\;$}}

    \UnaryInf$(p:P), (q:Q) \fCenter p:P$
    \UnaryInf$p:P \fCenter (\lambda q \ty Q.\; p) : Q \to P$
    \UnaryInf$\fCenter (\lambda p \ty P.\; \lambda q \ty Q.\; p) : P \to Q \to P$
    \BinaryInf$\fCenter (\lambda p \ty P.\; \lambda q \ty Q.\; p)\ x: Q \to P$
\end{prooftree}
